### 线性表

CLR中的线性表

c# 2.0 提供了泛型的IList<T>接口，实现了IList<T>接口的类有List<T>

------

使用CLR中的

```c#
//使用BLC中提供的List线性表

List<int> intList = new List<int>();

intList.Add(1);
intList.Add(2);
intList.Add(3);

intList.Remove(2);
intList.Clear();

Console.WriteLine(intList.Count);
Console.WriteLine(intList[1]);
```

定义线性表接口

```c#
interface IlistStruct<T>
    {
        int GetLength();
        void Clear();
        bool IsEmpty();
        T this[int index] { get; }
        T GetElem(int index);
        void Add(T item);
        void Insert(T item, int index);
        T Delete(int index);
        int Locate(T value);
    }
```

使用数组实现顺序表

```c#
 class SeqList<T> : IlistStruct<T>
    {
        private T[] data;   //存储数据
        private int count = 0;  //存了多少数据

        //构造函数
        public SeqList(int size)    //指定最大容量
        {
            data = new T[size];
            count = 0;
        }
        public SeqList():this(10)   //默认构造函数，容量10
        {

        }
        public T this[int index]
        {
            get
            {
                return GetElem(index);
            }
        }

        public void Add(T item)
        {
            if (count == data.Length)
            {
                Console.WriteLine("当前顺序表已存满,不允许再存");
            }
            else
            {
                data[count] = item;
                count++;
            }
        }

        public void Clear()
        {
            count = 0;
        }

        public T Delete(int index)
        {
            //找出返回指定位置的数
            //遍历索引后面的，位置前移动1
            //索引位置数被提走，从后面第一个开始前移

            T temp = data[index];
            for (int i = index+1; i < count; i++)
            {
                data[i - 1] = data[i];
            }
            count--;
            return temp;
        }

        public T GetElem(int index)
        {
            if (index >= 0 && index <= count-1)     //索引存在
            {
                return data[index];
            }
            else
            {
                Console.WriteLine("索引不存在");
                return default(T);
            }
        }

        public int GetLength()
        {
            return count;
        }

        public void Insert(T item, int index)
        {
            for (int i = count-1; i >= index; i--)
            {
                data[i + 1] = data[i];
            }
            data[index] = item;
            count++;
        }

        public bool IsEmpty()
        {
            return count == 0;
        }

        public int Locate(T value)
        {
            for (int i = 0; i < count; i++)
            {
                if (data[i].Equals(value))
                {
                    return i;
                }
            }
            return -1;
        }
    }
```

使用单链表实现

定义Node节点类

```c#
class Node<T>
    {
        //设置私有字段
        private T data;
        private Node<T> next;

        //构造函数
        public Node()
        {
            data = default(T);
            next = null;
        }

        public Node(T value)
        {
            data = value;
            next = null;
        }
        public Node(T value, Node<T> next)
        {
            this.data = value;
            this.next = next;
        }
        public Node(Node<T> next)
        {
            this.next = next;
        }

        //属性
        public T Data
        {
            get { return data; }
            set { data = value; }
        }

        public Node<T> Next
        {
            get { return next; }
            set { next = value; }
        }
    }
```

链表

```c#
class LinkList<T> : IlistStruct<T>
    {
        private Node<T> head;   //存储一个头节点
        public T this[int index]
        {
            get
            {
                Node<T> temp = head;
                for (int i = 1; i < index; i++)
                {
                    //让temp向后移动一个位置
                    temp = temp.Next;
                }
                return temp.Data;
            }
        }

        public void Add(T item)
        {
            //根据新数据创建一个新的节点
            Node<T> newNode = new Node<T>(item);
            //如果头节点为空，那么这个新节点就是头节点
            if (head == null)
            {
                head = newNode;
            }
            //把新来的节点放到链表的尾部
            //要访问到链表的尾节点
            else
            {
                Node<T> temp = head;
                while(true)
                {
                    if (temp.Next != null)
                    {
                        temp = temp.Next;
                    }
                    else
                    {
                        break;
                    }
                }
                //把新来的节点放到链表的尾部
                temp.Next = newNode;
            }
        }

        public void Clear()
        {
            head = null;
        }

        public T Delete(int index)
        {
            T data = default(T);
            //删除头节点
            if (index == 0)
            {
                data = head.Data;
                head = head.Next;
            }
            else
            {
                Node<T> temp = head;
                for (int i = 1; i < index-1; i++)
                {
                    //让temp向后移动一个位置
                    temp = temp.Next;
                }
                Node<T> preNode = temp;
                Node<T> currentNode = temp.Next;
                data = currentNode.Data;
                Node<T> nextNode = temp.Next.Next;
                preNode.Next = nextNode;
            }
            return data;
        }

        public T GetElem(int index)
        {
            return this[index];
        }

        public int GetLength()
        {
            if (head == null)
            {
                return 0;
            }

            Node<T> temp = head;
            int count = 1;
            while(true)
            {
                if (temp.Next != null)
                {
                    count++;
                    temp = temp.Next;
                }
                else
                {
                    break;
                }
            }
            return count;
        }

        public void Insert(T item, int index)
        {
            Node<T> newNode = new Node<T>(item);
            //插入到头节点
            if (index == 0)
            {
                newNode.Next = head;
                head = newNode;
            }
            //插入其他位置
            else
            {
                Node<T> temp = head;
                for (int i = 1; i <= index-1; i++)
                {
                    //让temp向后移动一位
                    temp = temp.Next;

                }
                Node<T> preNode = temp;
                Node<T> currentNode = temp.Next;
                preNode.Next = newNode;
                newNode.Next = currentNode;
            }
        }

        public bool IsEmpty()
        {
            return head == null;
        }

        public int Locate(T value)
        {
            Node<T> temp = head;
            if (temp == null)
            {
                return -1;
            }
            else
            {
                int index = 0;
                while (true)
                {
                    if (temp.Data.Equals(value))
                    {
                        return index;
                    }
                    else
                    {
                        if (temp.Next != null)
                        {
                            temp = temp.Next;
                        }
                        else
                        {
                            break;
                        }
                    }
                }
                return -1;
            }
        }
    }
```
