### 线性表

CLR中的线性表

c# 2.0 提供了泛型的IList<T>接口，实现了IList<T>接口的类有List<T>

------

使用CLR中的

```c#
//使用BLC中提供的List线性表

List<int> intList = new List<int>();

intList.Add(1);
intList.Add(2);
intList.Add(3);

intList.Remove(2);
intList.Clear();

Console.WriteLine(intList.Count);
Console.WriteLine(intList[1]);
```

定义线性表接口

```c#
interface IlistStruct<T>
    {
        int GetLength();
        void Clear();
        bool IsEmpty();
        T this[int index] { get; }
        T GetElem(int index);
        void Add(T item);
        void Insert(T item, int index);
        T Delete(int index);
        int Locate(T value);
    }
```

使用数组实现顺序表

```c#
 class SeqList<T> : IlistStruct<T>
    {
        private T[] data;   //存储数据
        private int count = 0;  //存了多少数据

        //构造函数
        public SeqList(int size)    //指定最大容量
        {
            data = new T[size];
            count = 0;
        }
        public SeqList():this(10)   //默认构造函数，容量10
        {

        }
        public T this[int index]
        {
            get
            {
                return GetElem(index);
            }
        }

        public void Add(T item)
        {
            if (count == data.Length)
            {
                Console.WriteLine("当前顺序表已存满,不允许再存");
            }
            else
            {
                data[count] = item;
                count++;
            }
        }

        public void Clear()
        {
            count = 0;
        }

        public T Delete(int index)
        {
            //找出返回指定位置的数
            //遍历索引后面的，位置前移动1
            //索引位置数被提走，从后面第一个开始前移

            T temp = data[index];
            for (int i = index+1; i < count; i++)
            {
                data[i - 1] = data[i];
            }
            count--;
            return temp;
        }

        public T GetElem(int index)
        {
            if (index >= 0 && index <= count-1)     //索引存在
            {
                return data[index];
            }
            else
            {
                Console.WriteLine("索引不存在");
                return default(T);
            }
        }

        public int GetLength()
        {
            return count;
        }

        public void Insert(T item, int index)
        {
            for (int i = count-1; i >= index; i--)
            {
                data[i + 1] = data[i];
            }
            data[index] = item;
            count++;
        }

        public bool IsEmpty()
        {
            return count == 0;
        }

        public int Locate(T value)
        {
            for (int i = 0; i < count; i++)
            {
                if (data[i].Equals(value))
                {
                    return i;
                }
            }
            return -1;
        }
    }
```

自定义顺序表的使用

```c#
//使用自己的顺序表
            SeqList<string> seqList = new SeqList<string>();

            seqList.Add("123");
            seqList.Add("456");
            seqList.Add("789");

            Console.WriteLine(seqList.GetElem(0));
            Console.WriteLine(seqList[0]);
            Console.WriteLine();
            seqList.Insert("666", 1);
            for (int i = 0; i < seqList.GetLength(); i++)
            {
                Console.WriteLine(seqList[i]+" ");
            }
            Console.WriteLine();
            seqList.Delete(1);
            for (int i = 0; i < seqList.GetLength(); i++)
            {
                Console.WriteLine(seqList[i] + " ");
            }
            Console.WriteLine();
            seqList.Clear();
            Console.WriteLine(seqList.GetLength());
```
