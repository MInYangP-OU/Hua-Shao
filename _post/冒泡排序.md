static void BubbleSort(int[] dataArray)
        {
            //控制回合
            for (int i = 0; i < dataArray.Length; i++)
            {
                //每一轮的冒泡处理
                for (int j = 0; j < dataArray.Length-i-1; j++)
                {
                    //Length-i-1;否则没有j+1
                    if (dataArray[j] > dataArray[j+1])
                    {
                        int iValue = dataArray[j+1];
                        dataArray[j + 1] = dataArray[j];
                        dataArray[j] = iValue;

                    }
                }
            }
        }
